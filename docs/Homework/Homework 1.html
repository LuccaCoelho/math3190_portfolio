#| label: gghist
gghist = function(x, binwidth = 1) {
  
  df = tibble(value = x)
  
  p = ggplot(data = df, aes(x = value)) +
    geom_histogram(binwidth = binwidth,fill = "steelblue", color = "black")

  return(p)
}
#| label: ggbox
#| echo: TRUE
#| eval: TRUE

# we need at least x
ggbox = function(x, y = NULL, direction = "horizontal") {
  
  direction = tolower(direction)
  
  # check if direction is the right input
  if (!direction %in% c("horizontal", "vertical")) {
    stop("Direction have to be either 'horizontal' or 'vertical'.")
  }
  
  # only numeric and no y variable
  if (is.numeric(x) && is.null(y)) {
    df = tibble(value = x)
    
    p = ggplot(data = df, aes(x = value)) +
          geom_boxplot(color = "black", fill = "steelblue", alpha = 0.6)
  }else if (is.numeric(x) && !is.null(y) && !is.numeric(y)) {
    
    df = tibble(
      value = x,
      group = as.factor(y)
    )
    
    p = ggplot(data = df, aes(x = value, y = group)) +
          geom_boxplot(color = "black", fill = "steelblue", alpha = 0.6)
  } else {
    stop("Provide either 1 numeric vector, or a numeric and a categorical vectors.")
  }
  
   if (direction == "vertical") {
    p <- p + coord_flip()
    }
  
  return(p)
}

create_team_summary = function(team_name) {
  
  initial_summary = cbbga_2025 |>
    mutate(winLoss = case_when((away_team == team & difference_scores < 0) ~ 1,
